#include <SFML/Graphics.hpp>

int main() {
	
	const int fieldWidth = 10;
	const int fieldHeight = 20;
	
	sf::Texture yellowTexture;
	if (!yellowTexture.loadFromFile("yellow.png")) {
		return EXIT_FAILURE;
	}
	const int blockWidth = yellowTexture.getSize().x;
	const int blockHeight = yellowTexture.getSize().y;
	const int windowWidth = fieldWidth * blockWidth;
	const int windowHeight = fieldHeight * blockHeight;
	sf::Sprite yellowSprite(yellowTexture);
	sf::RenderWindow window(sf::VideoMode(windowWidth, windowHeight), "Tetris");
	while (window.isOpen()) {
		sf::Event evt;
		if (window.pollEvent(evt)) {
			if (evt.type == sf::Event::Closed) {
				window.close();
			}
		}
	
		window.clear();
		window.draw(yellowSprite);
		window.display();
	
		
	}
	
	return EXIT_SUCCESS;

}

And we want to leave a record

  • use a vector to store status of field
#include <SFML/Graphics.hpp>

int main() {
	
	const int fieldWidth = 10;
	const int fieldHeight = 20;

	bool field[fieldWidth][fieldHeight] = {};

	
	sf::Texture yellowTexture;
	if (!yellowTexture.loadFromFile("yellow.png")) {
		return EXIT_FAILURE;
	}
	const int blockWidth = yellowTexture.getSize().x;
	const int blockHeight = yellowTexture.getSize().y;
	const int windowWidth = fieldWidth * blockWidth;
	const int windowHeight = fieldHeight * blockHeight;
	const sf::Vector2i origin(fieldWidth/2,0);
	sf::Vector2i pos(origin);

	sf::Sprite yellowSprite(yellowTexture);
	sf::RenderWindow window(sf::VideoMode(windowWidth, windowHeight), "Tetris");
	sf::Clock clock;
	while (window.isOpen()) {
		enum class Action {
			Hold,
			MoveDown,
			MoveLeft,
			MoveRight,
		};
		Action action = Action::Hold;
		sf::Event evt;
		if (window.pollEvent(evt)) {
			if (evt.type == sf::Event::Closed) {
				window.close();
			}
			if (evt.type == sf::Event::KeyPressed) {
				switch (evt.key.code) {
				case sf::Keyboard::Left:
					action = Action::MoveLeft;
					break;
				case sf::Keyboard::Right:
					action = Action::MoveRight;
					break;
				}
			}
		}
		sf::Vector2i nextPos(pos);
		if (clock.getElapsedTime().asSeconds() >= 0.3f) {
			action = Action::MoveDown;
			clock.restart();
		}
		switch (action) {
		case Action::Hold:
			break;
		case Action::MoveDown:
			nextPos.y++;
			break;
		case Action::MoveLeft:
			nextPos.x--;
			break;
		case Action::MoveRight:
			nextPos.x++;
			break;
		}

		//Also need to check nextPos has a block or not.
		if (nextPos.x > 0 && nextPos.x < fieldWidth && nextPos.y < fieldHeight 
			&& field[nextPos.x][nextPos.y] == false) {
			pos = nextPos;
		}// if nextPos is not vaild
		else {
			if (action == Action::MoveDown) {
				field[pos.x][pos.y] = true; // when we touch bottom, record it as true.
				pos = origin;  //let pos become origin which means move to beginning, when hit the bottom.
			}
	
		}

		window.clear();
		// draw field if field[][] is true
		for (int x = 0; x < fieldWidth; x++) {
			for (int y = 0; y < fieldHeight; y++) {
				if (field[x][y]) {
					yellowSprite.setPosition(
						float(x* blockWidth),
						float(y* blockHeight));
					window.draw(yellowSprite);
				}
			}
		}

		yellowSprite.setPosition(float(pos.x * blockWidth),
								float(pos.y * blockHeight));
		window.draw(yellowSprite);
		window.display();
	
		
	}
	
	return EXIT_SUCCESS;

}

#include <SFML/Graphics.hpp>
#include <vector>
int main() {
	
	const int fieldWidth = 10;
	const int fieldHeight = 20;

	bool field[fieldWidth][fieldHeight] = {};

	sf::Texture yellowTexture;
	if (!yellowTexture.loadFromFile("yellow.png")) {
		return EXIT_FAILURE;
	}
	// O:   OO  (0,-1) (1,-1)
	//      OO  (0, 0) (1, 0)
	std::vector<sf::Vector2i> shape_O = {
		sf::Vector2i(0,0),
		sf::Vector2i(1,0),
		sf::Vector2i(0,-1),
		sf::Vector2i(1,-1),
	};
	// I:  OOOO (-1,0) (0, 0) (1, 0) (2, 0) 
	std::vector<sf::Vector2i> shape_I = {
		sf::Vector2i(-1, 0),
		sf::Vector2i(0, 0),
		sf::Vector2i(1, 0),
		sf::Vector2i(2, 0),
	};

	std::vector<sf::Vector2i> currentShape;
	if (rand() % 2 == 0) {
		currentShape = shape_O;
	}
	else {
		currentShape = shape_I;
	}

	const int blockWidth = yellowTexture.getSize().x;
	const int blockHeight = yellowTexture.getSize().y;
	const int windowWidth = fieldWidth * blockWidth;
	const int windowHeight = fieldHeight * blockHeight;
	const sf::Vector2i origin(fieldWidth/2,0);
	sf::Vector2i pos(origin);

	sf::Sprite yellowSprite(yellowTexture);
	sf::RenderWindow window(sf::VideoMode(windowWidth, windowHeight), "Tetris");
	sf::Clock clock;
	while (window.isOpen()) {
		enum class Action {
			Hold,
			MoveDown,
			MoveLeft,
			MoveRight,
		};
		Action action = Action::Hold;
		sf::Event evt;
		if (window.pollEvent(evt)) {
			if (evt.type == sf::Event::Closed) {
				window.close();
			}
			if (evt.type == sf::Event::KeyPressed) {
				switch (evt.key.code) {
				case sf::Keyboard::Left:
					action = Action::MoveLeft;
					break;
				case sf::Keyboard::Right:
					action = Action::MoveRight;
					break;
				case sf::Keyboard::Down:
					action = Action::MoveDown;
					break;
				}
			}
		}
		sf::Vector2i nextPos(pos);
		if (clock.getElapsedTime().asSeconds() >= 0.3f) {
			action = Action::MoveDown;
			clock.restart();
		}
		switch (action) {
		case Action::Hold:
			break;
		case Action::MoveDown:
			nextPos.y++;
			break;
		case Action::MoveLeft:
			nextPos.x--;
			break;
		case Action::MoveRight:
			nextPos.x++;
			break;
		}
		// detection 
		int countEmpty = 0;
		bool isEmpty = false;
		for (const sf::Vector2i d : currentShape) {
			sf::Vector2i np = nextPos +d ;
			if (np.x >= 0 && np.x < fieldWidth &&
				np.y < fieldHeight &&
				(np.y < 0 || field[np.x][np.y] == false)) {
				countEmpty++;
			}
		}
		if (countEmpty == 4) {
			pos = nextPos;
		}
		else 
		{
			if (action == Action::MoveDown) {
				for (const sf::Vector2i& d : currentShape) {
					sf::Vector2i np = pos + d;
					if (np.y >= 0) {
						field[np.x][np.y] = true;
					}				
				}
				pos = origin;
				//when hit the bottom , randomly pick a shape
				if (rand() % 2 == 0) {
					currentShape = shape_O;
				}
				else {
					currentShape = shape_I;
				}
			}
		}
		

		window.clear();
		// draw field if field[][] is true
		for (int x = 0; x < fieldWidth; x++) {
			for (int y = 0; y < fieldHeight; y++) {
				if (field[x][y]) {
					yellowSprite.setPosition(
						float(x* blockWidth),
						float(y* blockHeight));
					window.draw(yellowSprite);
				}
			}
		}
		// draw a controling object
		// O:   OO  (0,-1) (1,-1)
        //      OO  (0, 0) (1, 0)      (0,0) is current pos

		for (const sf::Vector2i& d : currentShape) {
			sf::Vector2i np = pos + d;
			yellowSprite.setPosition(float(np.x* blockWidth),
				float(np.y* blockHeight));
			window.draw(yellowSprite);
			
		
		}
		window.display();
	
		
	}
	
	return EXIT_SUCCESS;

}
