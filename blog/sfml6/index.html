current source code

#include <SFML/Graphics.hpp>
#include <vector>
int main() {
	
	const int fieldWidth = 10;
	const int fieldHeight = 20;

	enum class Type {
		O,
		I,

	};
	bool field[fieldWidth][fieldHeight] = {};

	sf::Texture yellowTexture;
	if (!yellowTexture.loadFromFile("yellow.png")) {
		return EXIT_FAILURE;
	}

	sf::Texture lightBlueTexture;
	if (!lightBlueTexture.loadFromFile("light_blue.png")) {
		return EXIT_FAILURE;
	}
	// O:   OO  (0,-1) (1,-1)
	//      OO  (0, 0) (1, 0)
	std::vector<sf::Vector2i> shape_O = {
		sf::Vector2i(0,0),
		sf::Vector2i(1,0),
		sf::Vector2i(0,-1),
		sf::Vector2i(1,-1),
	};
	// I:  OOOO (-1,0) (0, 0) (1, 0) (2, 0) 
	std::vector<sf::Vector2i> shape_I = {
		sf::Vector2i(-1, 0),
		sf::Vector2i(0, 0),
		sf::Vector2i(1, 0),
		sf::Vector2i(2, 0),
	};

	Type currentType = Type(rand() % 2);
	
	
	const int blockWidth = yellowTexture.getSize().x;
	const int blockHeight = yellowTexture.getSize().y;
	const int windowWidth = fieldWidth * blockWidth;
	const int windowHeight = fieldHeight * blockHeight;
	const sf::Vector2i origin(fieldWidth/2,0);
	sf::Vector2i pos(origin);

	sf::Sprite yellowSprite(yellowTexture);
	sf::Sprite lightBlueSprite(lightBlueTexture);
	sf::Sprite currentSprite;
	// shape decides by different type
	std::vector<sf::Vector2i> currentShape;


	sf::RenderWindow window(sf::VideoMode(windowWidth, windowHeight), "Tetris");
	sf::Clock clock;
	while (window.isOpen()) {
		switch (currentType) {
		case Type::O:
			currentShape = shape_O;
			currentSprite = yellowSprite;
			break;
		case Type::I:
			currentShape = shape_I;
			currentSprite = lightBlueSprite;
			break;
		
		}
		enum class Action {
			Hold,
			MoveDown,
			MoveLeft,
			MoveRight,
		};
		Action action = Action::Hold;
		sf::Event evt;
		if (window.pollEvent(evt)) {
			if (evt.type == sf::Event::Closed) {
				window.close();
			}
			if (evt.type == sf::Event::KeyPressed) {
				switch (evt.key.code) {
				case sf::Keyboard::Left:
					action = Action::MoveLeft;
					break;
				case sf::Keyboard::Right:
					action = Action::MoveRight;
					break;
				case sf::Keyboard::Down:
					action = Action::MoveDown;
					break;
				}
			}
		}
		sf::Vector2i nextPos(pos);
		if (clock.getElapsedTime().asSeconds() >= 0.3f) {
			action = Action::MoveDown;
			clock.restart();
		}
		switch (action) {
		case Action::Hold:
			break;
		case Action::MoveDown:
			nextPos.y++;
			break;
		case Action::MoveLeft:
			nextPos.x--;
			break;
		case Action::MoveRight:
			nextPos.x++;
			break;
		}
		// detection 
		int countEmpty = 0;
		bool isEmpty = false;
		for (const sf::Vector2i d : currentShape) {
			sf::Vector2i np = nextPos +d ;
			if (np.x >= 0 && np.x < fieldWidth &&
				np.y < fieldHeight &&
				(np.y < 0 || field[np.x][np.y] == false)) {
				countEmpty++;
			}
		}
		if (countEmpty == 4) {
			pos = nextPos;
		}
		else 
		{
			if (action == Action::MoveDown) {
				for (const sf::Vector2i& d : currentShape) {
					sf::Vector2i np = pos + d;
					if (np.y >= 0) {
						field[np.x][np.y] = true;
					}				
				}
				pos = origin;
				//when hit the bottom , randomly pick a shape
				currentType = Type(rand() % 2);
			}
		}
		

		window.clear();
		// draw field if field[][] is true
		for (int x = 0; x < fieldWidth; x++) {
			for (int y = 0; y < fieldHeight; y++) {
				if (field[x][y]) {
					currentSprite.setPosition(
						float(x* blockWidth),
						float(y* blockHeight));
					window.draw(currentSprite);
				}
			}
		}
		// draw a controling object
		// O:   OO  (0,-1) (1,-1)
        //      OO  (0, 0) (1, 0)      (0,0) is current pos

		for (const sf::Vector2i& d : currentShape) {
			sf::Vector2i np = pos + d;
			currentSprite.setPosition(float(np.x* blockWidth),
				float(np.y* blockHeight));
			window.draw(currentSprite);
			
		
		}
		window.display();
	
		
	}
	
	return EXIT_SUCCESS;

}

Remain the block color which is stored in the field

  • bool field[fieldWidth][fieldHeight] = {}; bool only represent True or False
  • Now, we care what we put in the field
enum class Type {
		None,
		O,
		I,

	};
	Type field[fieldWidth][fieldHeight] = {};
  • Checking part we need to change
if (np.x >= 0 && np.x < fieldWidth &&
				np.y < fieldHeight &&
				(np.y < 0 || field[np.x][np.y] == Type::None)) {
				countEmpty++;
			}
  • Draw field
// draw field if field[][] is true
		for (int x = 0; x < fieldWidth; x++) {
			for (int y = 0; y < fieldHeight; y++) {
				if (field[x][y] == Type::None) continue;
				switch((field[x][y]){
                case Type::O:
                    yellowSprite.setPosition(
					float(x* blockWidth),
					float(y* blockHeight));
					window.draw(yellowSprite);
                    break;
                case Type::I:
                    lightBlueSprite.setPosition(
					float(x* blockWidth),
					float(y* blockHeight));
					window.draw(lightBlueSprite);
                    break;
				}

                }
					
			}

                    break;
				}

					window.draw(s);
                }
					
			}
