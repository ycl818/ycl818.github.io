呼叫函式

#include <SFML/Window.hpp>
int main(){
    
    sf::Window w(sf::VideoMode(400, 200), "Hello");    

    while(w.isOpen()){
        sf::Event evt;  //參考不能傳匿名的物件
       if (w.pollEvent(evt)) {
            //evt 存放著使用者互動發生的事件
            if (evt.type == sf::Event::Closed) {
                std::cout << "close" << std::endl;
                w.close();
            }
        }
        w.display();
    }
    return 0;
}

#include <SFML/Window.hpp>
int main(){
    
    sf::RebderWindow w(sf::VideoMode(400, 200), "Hello");    
    sf::Font font;
    
    if(font.loadFromFile("arial.ttf") == false ){
        return 1;

    sf::Text text("Hello", font);  //文字也是一次

   return EXIT_SUCCESS;
}

#include <SFML/Window.hpp>
int main(){
    
    sf::RebderWindow w(sf::VideoMode(400, 200), "Hello");    
    sf::Font font;
    
    if(font.loadFromFile("arial.ttf") == false ){
        return 1;
    }                               //優化 字形只要產生一次不然每輪迴圈都要產生一次
    sf::Text text("Hello", font);  //文字也是一次

    while(w.isOpen()){
        sf::Event evt;  //參考不能傳匿名的物件
       if (w.pollEvent(evt)) {
            //evt 存放著使用者互動發生的事件
            if (evt.type == sf::Event::Closed) {
                std::cout << "close" << std::endl;
                w.close();
            }
        }
        w.clear(); // 清掉
        w.draw(text);
        w.display();

        text.move(0.01,0);   // move 吃float
    }
   return EXIT_SUCCESS;
}
